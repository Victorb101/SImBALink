\section{Environment} This system models the environment of the motorcycle is riding in.

\begin{gather}
given: h(d) \\
\theta_r  = \frac{d}{dd} h(d) \\
T_amb(d) = T_0 - L h(d) \\
P(d) = P_0 \left( 1 - \frac{L h(d)}{T_0} \right)^{\frac{gM}{RL}}\\
\rho(d) = \frac{PM}{1000RT}  
\end{gather}
